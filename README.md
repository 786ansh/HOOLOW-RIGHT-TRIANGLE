# HOLLOW-RIGHT-TRIANGLE

int main()
{
    for(int i=0;i<6;i++)
    {
        for(int j=0;j<6;j++)
        {
            if(i==5||j==0||i==j)
            {
                printf("*");
                
            }
          
            else 
            printf(" ");


        }
        printf("\n");
    }

    return 0;
}
